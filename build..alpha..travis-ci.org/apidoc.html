function koa-helmet ()

  • description and source-code
    koa-helmet = function () {
  const helmetPromise = promisifyMiddleware(helmet.apply(null, arguments))

  return (ctx, next) => {
    ctx.req.secure = ctx.request.secure
    return helmetPromise(ctx.req, ctx.res).then(next)
  }
}
